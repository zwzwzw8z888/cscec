#!/usr/bin/env python
# -*- coding: utf-8 -*-
"""读取 数据输入/ 下所有 Excel，返回结构化数据"""

import os
from pathlib import Path
import openpyxl

BASE_DIR = Path(__file__).parent
INPUT_DIR = BASE_DIR / "数据输入"


def read_all_excels():
    """扫描输入目录，读取所有 .xlsx 文件。

    Returns:
        dict: {文件名: {sheet名: [list of rows]}}
        每行是一个 dict，key 为列头（第1行），value 为单元格值
    """
    result = {}
    if not INPUT_DIR.exists():
        print(f"[警告] 输入目录不存在: {INPUT_DIR}")
        return result

    xlsx_files = list(INPUT_DIR.glob("*.xlsx"))
    if not xlsx_files:
        print(f"[警告] 输入目录为空，请放入 .xlsx 文件: {INPUT_DIR}")
        return result

    for fp in xlsx_files:
        print(f"  读取: {fp.name}")
        wb = openpyxl.load_workbook(str(fp), data_only=True)
        sheets = {}
        for sheet_name in wb.sheetnames:
            ws = wb[sheet_name]
            rows = []
            headers = []
            for row_idx, row in enumerate(ws.iter_rows(values_only=True), start=1):
                if row_idx == 1:
                    headers = [str(c).strip() if c else "" for c in row]
                    # 跳过空表头或无数据sheet
                    if not any(headers):
                        continue
                else:
                    row_dict = {}
                    for i, val in enumerate(row):
                        if i < len(headers) and headers[i]:
                            row_dict[headers[i]] = val
                    if row_dict:  # 跳过全空行
                        rows.append(row_dict)
            if rows:
                sheets[sheet_name] = rows
        result[fp.name] = sheets
        wb.close()

    print(f"  共读取 {len(result)} 个文件")
    return result


def filter_empty_sheets(dataframes):
    """过滤掉无数据的 sheet，保留有效数据。

    Returns:
        dict: 仅包含有数据行的 {文件名: {sheet名: [rows]}}
    """
    result = {}
    for fname, sheets in dataframes.items():
        valid_sheets = {}
        for sname, rows in sheets.items():
            if rows:
                valid_sheets[sname] = rows
        if valid_sheets:
            result[fname] = valid_sheets
    return result
```

#!/usr/bin/env python
# -*- coding: utf-8 -*-
"""SQLite 数据存储 —— 月度快照读写"""

import json
import sqlite3
from pathlib import Path

DB_PATH = Path(__file__).parent / "data.db"


def _connect():
    conn = sqlite3.connect(str(DB_PATH))
    conn.row_factory = sqlite3.Row
    return conn


def init_db():
    """初始化数据库表"""
    conn = _connect()
    conn.execute("""
        CREATE TABLE IF NOT EXISTS snapshots (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            year_month TEXT NOT NULL,
            section TEXT NOT NULL,
            unit_name TEXT NOT NULL,
            metrics TEXT NOT NULL DEFAULT '{}',
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            UNIQUE(year_month, section, unit_name)
        )
    """)
    conn.commit()
    conn.close()


def save_snapshot(year_month, section, unit_name, metrics):
    """保存或更新一条快照记录。

    Args:
        year_month: '2026-04'
        section: '新开工项目上线'
        unit_name: '一公司' or '合计'
        metrics: dict of {指标名: 值}
    """
    init_db()
    conn = _connect()
    conn.execute("""
        INSERT OR REPLACE INTO snapshots (year_month, section, unit_name, metrics)
        VALUES (?, ?, ?, ?)
    """, (year_month, section, unit_name, json.dumps(metrics, ensure_ascii=False)))
    conn.commit()
    conn.close()


def get_snapshot(year_month, section, unit_name):
    """获取单条快照。

    Returns:
        dict or None: {指标名: 值}
    """
    init_db()
    conn = _connect()
    row = conn.execute(
        "SELECT metrics FROM snapshots WHERE year_month=? AND section=? AND unit_name=?",
        (year_month, section, unit_name)
    ).fetchone()
    conn.close()
    if row:
        return json.loads(row["metrics"])
    return None


def get_section_snapshot(year_month, section):
    """获取某板块所有单位的快照。

    Returns:
        dict: {unit_name: {指标名: 值}}
    """
    init_db()
    conn = _connect()
    rows = conn.execute(
        "SELECT unit_name, metrics FROM snapshots WHERE year_month=? AND section=?",
        (year_month, section)
    ).fetchall()
    conn.close()
    result = {}
    for row in rows:
        result[row["unit_name"]] = json.loads(row["metrics"])
    return result


def has_previous_month(year_month):
    """检查是否有上月数据"""
    from constants import prev_month
    prev = prev_month(year_month)
    init_db()
    conn = _connect()
    count = conn.execute(
        "SELECT COUNT(*) as cnt FROM snapshots WHERE year_month=?", (prev,)
    ).fetchone()["cnt"]
    conn.close()
    return count > 0
```

#!/usr/bin/env python
# -*- coding: utf-8 -*-
"""
配方：新开工项目当月上线情况
数据源：项目基本信息台账.xlsx → 项目信息台账 sheet
逻辑：
  1. 筛选 工程项目状态 包含"施工准备"或"正常施工" → 新开工项目
  2. 筛选 系统应用阶段 == "已上线" → 已上线
  3. 按 所属末级管理责任主体 分组统计
  4. 计算比例
"""

from config import map_unit_name, get_unit_sort_key
from constants import UNITS_ORDER


def process(dataframes):
    """从数据中提取新开工项目上线情况。

    Args:
        dataframes: {文件名: {sheet名: [list of dict rows]}}

    Returns:
        dict: {
            'section_key': '推广应用_新开工项目上线',
            'section_title': '（二）新开工项目当月上线情况',
            'pre_text': '...引导文字...',
            'post_note': '...注释...',
            'table_headers': ['序号', '单位名称', ...],
            'table_data': [['1', '一公司', ...], ...],
            'snapshot_data': {'一公司': {'项目数量': 3, '已上线数量': 1, ...}, ...}
        }
    """
    # 找到项目基本信息台账的数据
    project_rows = []
    for fname, sheets in dataframes.items():
        if "项目基本信息台账" in fname or "项目信息台账" in fname:
            for sname, rows in sheets.items():
                project_rows.extend(rows)

    if not project_rows:
        print("  [新开工项目上线] 未找到项目基本信息数据，跳过")
        return None

    # 筛选新开工项目：工程项目状态包含"施工准备"或"正常施工"
    new_projects = []
    for row in project_rows:
        status = str(row.get("工程项目状态", "") or "")
        if "施工准备" in status or "正常施工" in status:
            new_projects.append(row)

    # 按单位分组统计
    unit_stats = {}
    for row in new_projects:
        org_name = row.get("所属末级管理责任主体", "")
        short_name = map_unit_name(org_name)
        if not short_name:
            continue

        if short_name not in unit_stats:
            unit_stats[short_name] = {"项目数量": 0, "已上线数量": 0}

        unit_stats[short_name]["项目数量"] += 1

        app_stage = str(row.get("系统应用阶段", "") or "")
        if "已上线" in app_stage:
            unit_stats[short_name]["已上线数量"] += 1

    # 计算比例
    for unit in unit_stats:
        n = unit_stats[unit]["项目数量"]
        u = unit_stats[unit]["已上线数量"]
        unit_stats[unit]["已上线比例"] = f"{u/n*100:.0f}%" if n > 0 else "/"

    # 按通报顺序排列 + 合计
    table_data = []
    seq = 0
    for unit in UNITS_ORDER:
        if unit in unit_stats:
            seq += 1
            s = unit_stats[unit]
            table_data.append([
                str(seq), unit,
                str(s["项目数量"]),
                str(s["已上线数量"]),
                s["已上线比例"],
            ])

    # 合计行
    total_count = sum(s["项目数量"] for s in unit_stats.values())
    total_online = sum(s["已上线数量"] for s in unit_stats.values())
    total_pct = f"{total_online/total_count*100:.0f}%" if total_count > 0 else "/"
    table_data.append([
        "", "总计",
        str(total_count),
        str(total_online),
        total_pct,
    ])

    unit_stats["合计"] = {
        "项目数量": total_count,
        "已上线数量": total_online,
        "已上线比例": total_pct,
    }

    return {
        "section_key": "推广应用_新开工项目上线",
        "section_title": "（二）新开工项目当月上线情况",
        "pre_text": (
            f"截止上月累计新开工项目X个，已上线X个。"
            f"本月新开工项目{total_count}个，已上线{total_online}个。详见下表："
        ),
        "post_note": (
            "注：本数据来源是根据局项目管理中心，每月收集并确认的新开工项目清单，"
            "结合开工即上线的目标要求进行统计计算得出。"
        ),
        "table_headers": ["序号", "单位名称", "项目数量", "已上线数量", "已上线比例"],
        "table_data": table_data,
        "snapshot_data": unit_stats,
    }
```

    for row in new_projects:
        org_name = row.get("所属末级管理责任主体", "")
        short_name = map_unit_name(org_name)
        if not short_name:
            continue

        if short_name not in unit_stats:
            unit_stats[short_name] = {"项目数量": 0, "已集成数量": 0}

        unit_stats[short_name]["项目数量"] += 1

        proj_code = str(row.get("工程项目编码", "") or row.get("中建四局项目编码", "") or "")
        if proj_code in integrated_codes:
            unit_stats[short_name]["已集成数量"] += 1

    # 计算比例 + 建表
    for unit in unit_stats:
        n = unit_stats[unit]["项目数量"]
        i = unit_stats[unit]["已集成数量"]
        unit_stats[unit]["已集成比例"] = f"{i/n*100:.1f}%" if n > 0 else "/"

    table_data = []
    seq = 0
    for unit in UNITS_ORDER:
        if unit in unit_stats:
            seq += 1
            s = unit_stats[unit]
            table_data.append([
                str(seq), unit,
                str(s["项目数量"]),
                str(s["已集成数量"]),
                s["已集成比例"],
            ])

    total_n = sum(s["项目数量"] for s in unit_stats.values())
    total_i = sum(s["已集成数量"] for s in unit_stats.values())
    total_pct = f"{total_i/total_n*100:.1f}%" if total_n > 0 else "/"
    table_data.append(["", "合计", str(total_n), str(total_i), total_pct])

    unit_stats["合计"] = {
        "项目数量": total_n, "已集成数量": total_i, "已集成比例": total_pct
    }

    return {
        "section_key": "推广应用_业财集成",
        "section_title": "（三）新开项目业财集成情况",
        "pre_text": (
            f"截止上月累计新开工项目X个，已集成X个。"
            f"本月新开工项目{total_n}个，已集成{total_i}个。详见下表："
        ),
        "post_note": (
            "注：项目数量来源是根据局项目管理中心，每月收集并确认的新开工项目清单，"

#!/usr/bin/env python
# -*- coding: utf-8 -*-
"""Word 文档拼装 —— 完整通报生成"""

from docx import Document
from docx.shared import Pt, Cm
from docx.enum.text import WD_ALIGN_PARAGRAPH
from docx.oxml.ns import qn
from docx.oxml import OxmlElement

from constants import (
    FONT_FANGSONG, FONT_HEITI, FONT_KAITI, FONT_XIAOBIAOSONG, FONT_TNR,
    SIZE_CHUHAO, SIZE_ERHAO, SIZE_SANHAO, SIZE_XIAOSI,
    LINE_SPACING_TWIPS, LINE_SPACING_HEADER,
    MARGIN_TOP, MARGIN_BOTTOM, MARGIN_LEFT, MARGIN_RIGHT,
    FIRST_LINE_INDENT,
    ALIGN_LEFT, ALIGN_CENTER, ALIGN_RIGHT, ALIGN_JUSTIFY,
    RED_HEADER_TEXT, LETTER_PREFIX, MAIN_TITLE_TEMPLATE,
    RECIPIENT, CC_TEXT, SIGNATURE_DEPT, END_PAGE_TEXT,
    month_to_cn,
)


def _set_font(run, cn_font, size_pt, bold=False, color=None):
    """设置 run 的中英文字体、字号"""
    run.font.size = Pt(size_pt)
    run.bold = bold
    run.font.name = cn_font
    r = run._element
    rPr = r.find(qn('w:rPr'))
    if rPr is None:
        rPr = OxmlElement('w:rPr')
        r.insert(0, rPr)
    rFonts = rPr.find(qn('w:rFonts'))
    if rFonts is None:
        rFonts = OxmlElement('w:rFonts')
        rPr.insert(0, rFonts)
    rFonts.set(qn('w:eastAsia'), cn_font)
    rFonts.set(qn('w:ascii'), FONT_TNR)
    rFonts.set(qn('w:hAnsi'), FONT_TNR)
    if color:
        run.font.color.rgb = color


def _set_line_spacing(paragraph, twips=LINE_SPACING_TWIPS):
    """设置段落行距为固定值"""
    pPr = paragraph._element.get_or_add_pPr()
    spacing = pPr.find(qn('w:spacing'))
    if spacing is None:
        spacing = OxmlElement('w:spacing')
        pPr.append(spacing)
    spacing.set(qn('w:line'), str(twips))
    spacing.set(qn('w:lineRule'), 'exact')


def _set_first_line_indent(paragraph, indent=0):
    """设置首行缩进"""
    pPr = paragraph._element.get_or_add_pPr()
    ind = pPr.find(qn('w:ind'))
    if ind is None:
        ind = OxmlElement('w:ind')
        pPr.append(ind)
    if indent > 0:
        ind.set(qn('w:firstLine'), str(indent))


def _set_justify(paragraph):
    paragraph.alignment = ALIGN_JUSTIFY


def add_text_paragraph(doc, text, cn_font, size_pt, bold=False, alignment=None,
                       first_line_indent=0, spacing=LINE_SPACING_TWIPS):
    """添加一个格式化文本段落"""
    p = doc.add_paragraph()
    if alignment is not None:
        p.alignment = alignment
    run = p.add_run(text)
    _set_font(run, cn_font, size_pt, bold=bold)
    _set_line_spacing(p, spacing)
    if first_line_indent:
        _set_first_line_indent(p, first_line_indent)
    return p


def add_table_to_doc(doc, headers, data, col_widths=None):
    """添加格式化表格。

    Args:
        doc: Document 对象
        headers: ['序号', '单位名称', ...]
        data: [['1', '一公司', ...], ...]
        col_widths: [Cm(1.2), Cm(3.5), ...] or None
    """
    table = doc.add_table(rows=1 + len(data), cols=len(headers))
    table.style = 'Table Grid'

    # 表头
    for i, h in enumerate(headers):
        cell = table.rows[0].cells[i]
        p = cell.paragraphs[0]
        p.alignment = ALIGN_CENTER
        run = p.add_run(h)
        _set_font(run, FONT_FANGSONG, SIZE_XIAOSI, bold=True)
        _set_line_spacing(p)

    # 数据行
    for r, row_data in enumerate(data):
        for c, val in enumerate(row_data):
            cell = table.rows[r + 1].cells[c]
            p = cell.paragraphs[0]
            p.alignment = ALIGN_CENTER
            run = p.add_run(str(val))
            _set_font(run, FONT_FANGSONG, SIZE_XIAOSI, bold=False)
            _set_line_spacing(p)

    # 设置列宽
    if col_widths:
        for row in table.rows:
            for i, width in enumerate(col_widths):
                row.cells[i].width = width

    return table
```

def build_report(output_path, year, month, sections_data):
    """拼装完整通报文档。

    Args:
        output_path: 输出 .docx 路径
        year: 2026
        month: 4
        sections_data: list of recipe results (dicts with section_key, section_title,
                       pre_text, post_note, table_headers, table_data)
    """

    # ── 页面设置 ──

    section.bottom_margin = MARGIN_BOTTOM

    section.right_margin = MARGIN_RIGHT

    add_text_paragraph(doc, letter_num, FONT_FANGSONG, SIZE_SANHAO,
                       alignment=ALIGN_CENTER)

    # ── 空行 ──
    add_text_paragraph(doc, "", FONT_FANGSONG, SIZE_SANHAO)

    # ── 标题 ──
    month_cn = month_to_cn(month)
    title = f"{MAIN_TITLE_TEMPLATE}（{year}年{month_cn}月）"
    add_text_paragraph(doc, title, FONT_XIAOBIAOSONG, SIZE_ERHAO,
                       bold=False, alignment=ALIGN_CENTER)

    # ── 主送单位 ──
    add_text_paragraph(doc, RECIPIENT, FONT_FANGSONG, SIZE_SANHAO,
                       alignment=ALIGN_LEFT)

    # ── 导语 ──
    intro = f"现将全局{year}年{month_cn}月份DMP推广应用情况通报如下："
    add_text_paragraph(doc, intro, FONT_FANGSONG, SIZE_SANHAO,
                       first_line_indent=FIRST_LINE_INDENT)

    # ── 遍历板块 ──
    current_section = None
    for sec in sections_data:
        if sec is None:
            continue

        section_key = sec.get("section_key", "")

        # 一级标题（如"一、建设情况"）
        if section_key.startswith("建设") and current_section != "一":
            current_section = "一"
            add_text_paragraph(doc, "一、建设情况", FONT_HEITI, SIZE_SANHAO,
                               bold=False, first_line_indent=FIRST_LINE_INDENT)
        elif section_key.startswith("推广应用") and current_section != "二":
            current_section = "二"
            add_text_paragraph(doc, "二、推广应用情况", FONT_HEITI, SIZE_SANHAO,
                               bold=False, first_line_indent=FIRST_LINE_INDENT)
        elif section_key.startswith("运维") and current_section != "三":
            current_section = "三"
            add_text_paragraph(doc, "三、系统运维情况", FONT_HEITI, SIZE_SANHAO,
                               bold=False, first_line_indent=FIRST_LINE_INDENT)

        # 二级标题（如"（二）新开工项目当月上线情况"）
        section_title = sec.get("section_title", "")
        if section_title:
            add_text_paragraph(doc, section_title, FONT_KAITI, SIZE_SANHAO,
                               bold=False, first_line_indent=FIRST_LINE_INDENT)

        # 引导文字
        pre_text = sec.get("pre_text", "")
        if pre_text:
            add_text_paragraph(doc, pre_text, FONT_FANGSONG, SIZE_SANHAO,
                               first_line_indent=FIRST_LINE_INDENT)

        # 表格
        headers = sec.get("table_headers", [])
        data = sec.get("table_data", [])
        if headers and data:
            add_table_to_doc(doc, headers, data)

    add_text_paragraph(doc, "", FONT_FANGSONG, SIZE_SANHAO)  # 空行
    add_text_paragraph(doc, END_PAGE_TEXT, FONT_FANGSONG, SIZE_SANHAO,

    add_text_paragraph(doc, "", FONT_FANGSONG, SIZE_SANHAO)  # 空行

    date_str = f"{year}年{month}月X日"

    add_text_paragraph(doc, CC_TEXT, FONT_FANGSONG, SIZE_SANHAO,
                       alignment=ALIGN_LEFT)

    doc.save(output_path)
    print(f"  通报已生成: {output_path}")
